    2. أساسيات Python¶

    2.1 المتغيرات: تخزين المعلومات¶

    لماذا نحتاج المتغيرات؟¶

    عندما تحلل تسلسلاً جينياً، تحتاج مكاناً تحفظ فيه التسلسل، طوله، اسم الجين، نسبة GC... كل هذه قيم تحتاج أن تُخزَّن وتُسترجع. المتغير هو ببساطة اسم تعطيه لقيمة معينة حتى تستخدمها لاحقاً. مثل س في الرياضيات فهي خانة لرقم متغير، لكن في بايثون لتخزين قيمة والعودة لها.

    المفهوم الأساسي¶

    # المتغير = اسم يشير إلى قيمة
gene_name = "BRCA1"

    السطر أعلاه يقول: "خزّن النص BRCA1 تحت الاسم gene_name". الآن يمكنك استخدام gene_name في أي مكان بدلاً من كتابة "BRCA1" كل مرة. في كل مرة تذكر المتغير فأنت تذكر الجين.

    أنواع البيانات الأساسية¶

    كل قيمة في Python لها نوع يحدد ما يمكنك فعله بها:

    # نص (str) — لتخزين الأسماء والتسلسلات
gene_name = "BRCA1"
dna_sequence = "ATGCGA"

# عدد صحيح (int) — للعد والفهرسة
sequence_length = 5589
num_exons = 23

# عدد عشري (float) — للقياسات والنسب
gc_content = 55.6
p_value = 0.003

# قيمة منطقية (bool) — للأسئلة بنعم/لا
is_coding = True
has_mutation = False

    قواعد تسمية المتغيرات

    • استخدم أسماء وصفية: gene_name أفضل بكثير من g أو x
    • افصل الكلمات بشرطة سفلية: gc_content وليس gccontent
    • لا تبدأ الاسم برقم: seq1 صحيح، 1seq خطأ
    • تجنب الأسماء المحجوزة: لا تسمِّ متغيراً print أو list

    تجربة عملية¶

    # لنخزن معلومات عن جين TP53
gene_name = "TP53"
chromosome = 17
length_bp = 19149
is_tumor_suppressor = True

# يمكننا طباعة هذه المعلومات
print(gene_name)              # TP53
print(length_bp)              # 19149
print(is_tumor_suppressor)    # True

# يمكننا أيضاً دمجها في جملة
print(f"الجين {gene_name} يقع على الكروموسوم {chromosome}")
# الإخراج: الجين TP53 يقع على الكروموسوم 17

    دالة print() تعرض القيمة على الشاشة. حرف f قبل النص يعني "formatted string" — يسمح لك بإدراج قيم المتغيرات داخل النص باستخدام الأقواس المعقوفة {}.

    2.2 النصوص: التعامل مع التسلسلات¶

    لماذا النصوص مهمة؟¶

    تسلسلات DNA و RNA والبروتين هي نصوص بطبيعتها — سلسلة من الحروف. تسلسل DNA مثل ATGCGATCG هو نص مكون من 9 أحرف. لذلك فهم كيفية التعامل مع النصوص في Python يعني فهم كيفية التعامل مع التسلسلات البيولوجية.

    العمليات الأساسية على النصوص¶

    طول النص — كم نيوكليوتيد في التسلسل؟

    dna = "ATGCGATCG"
print(len(dna))  # 9

    دالة len() تُرجع عدد الحروف. في السياق البيولوجي، هذا يعني عدد القواعد النيتروجينية (bp).

    التقطيع — استخراج جزء من التسلسل:

    dna = "ATGCGATCGTAGC"

# أول 3 حروف (الكودون الأول)
first_codon = dna[0:3]
print(first_codon)  # ATG

# ثاني كودون
second_codon = dna[3:6]
print(second_codon)  # CGA

# آخر 3 حروف
last_codon = dna[-3:]
print(last_codon)  # AGC

    الفهرسة تبدأ من الصفر

    في Python، الحرف الأول ترتيبه 0 وليس 1. هذا مربك في البداية لكنه اتفاقية ثابتة:

    التسلسل:  A  T  G  C  G  A
الفهرس:   0  1  2  3  4  5

    لذلك dna[0] تعطيك A، و dna[0:3] تعطيك الحروف من الفهرس 0 إلى 2 (الرقم الأخير لا يُضمَّن).

    العد — كم مرة يظهر حرف معين؟

    dna = "ATGCGATCGTAGC"

# عد كل نيوكليوتيد
a_count = dna.count('A')  # 3
t_count = dna.count('T')  # 2
g_count = dna.count('G')  # 4
c_count = dna.count('C')  # 3

print(f"A={a_count}, T={t_count}, G={g_count}, C={c_count}")

    الاستبدال — تحويل DNA إلى RNA:

    dna = "ATGCGATCG"

# النسخ: استبدال T بـ U
rna = dna.replace('T', 'U')
print(rna)  # AUGCGAUCG

    .replace() تنشئ نسخة جديدة من النص مع الاستبدال — النص الأصلي لا يتغير.

    تطبيق عملي: حساب نسبة GC¶

    نسبة GC (GC content) هي من أهم المقاييس الأساسية في تحليل التسلسلات. تخبرنا عن استقرار الـ DNA لأن الروابط بين G و C أقوى (3 روابط هيدروجينية) من الروابط بين A و T (رابطتان).

    dna = "ATGCGATCGTAGC"

# الخطوة 1: عد G و C
g_count = dna.count('G')  # 4
c_count = dna.count('C')  # 3
gc_total = g_count + c_count  # 7

# الخطوة 2: حساب النسبة
total_bases = len(dna)  # 13
gc_percent = (gc_total / total_bases) * 100

print(f"GC content: {gc_percent:.1f}%")  # GC content: 53.8%

    :.1f في السطر الأخير يعني "اعرض رقماً عشرياً واحداً بعد النقطة".

    تطبيق: تصنيف التسلسلات حسب محتوى GC¶

    def classify_gc(sequence):
    """تصنيف تسلسل DNA حسب نسبة GC."""
    gc_count = sequence.count('G') + sequence.count('C')
    gc_percent = (gc_count / len(sequence)) * 100

    if gc_percent < 30:
        return "AT-rich (منخفض GC)"
    elif gc_percent < 45:
        return "AT-leaning (مائل لـ AT)"
    elif gc_percent < 55:
        return "Balanced (متوازن)"
    elif gc_percent < 70:
        return "GC-leaning (مائل لـ GC)"
    else:
        return "GC-rich (عالي GC)"

# أمثلة
print(classify_gc("AAATTTAAAT"))   # AT-rich (منخفض GC)
print(classify_gc("ATGCGATCGA"))   # GC-leaning (مائل لـ GC)
print(classify_gc("GCGCGCGCGC"))   # GC-rich (عالي GC)

    لماذا تهمنا نسبة GC؟

    • الكائنات الحية لها نسب GC مميزة (مثلاً: E. coli ~51%، Plasmodium ~20%)
    • المناطق الغنية بـ GC في الجينوم البشري غالباً تحتوي على كثافة أعلى من الجينات
    • نسبة GC تؤثر على درجة حرارة الانصهار (Tm) للـ DNA
    • نقل الجينات الأفقي يُكتشف أحياناً عبر اختلاف نسبة GC

    2.7 الدوال: بناء أدوات قابلة لإعادة الاستخدام¶

    لماذا نحتاج الدوال؟¶

    لاحظ أننا حسبنا نسبة GC أكثر من مرة بنفس الطريقة. بدلاً من نسخ نفس الأسطر كل مرة، نجمعها في دالة — مجموعة من الكود المسمى يمكنك استدعاؤه في أي وقت.

    الدوال تحقق ثلاثة أهداف:

    1. عدم التكرار — اكتب الكود مرة واحدة، استخدمه مراراً
    2. الوضوح — calculate_gc(seq) أوضح من 4 أسطر حسابية
    3. الموثوقية — إذا وجدت خطأ، تصلحه في مكان واحد

    بنية الدالة¶

    def calculate_gc(sequence):

    gc_count = sequence.count('G') + sequence.count('C')
    return (gc_count / len(sequence)) * 100

    • calculate_gc اسم الدالة
    • sequence هو المعامل — القيمة التي تدخل إلى الدالة
    • النص بين """ يسمى docstring — يشرح ما تفعله الدالة
    • return تُرجع النتيجة

    دالة أكثر تعقيداً: المتمم العكسي¶

    المتمم العكسي (reverse complement) هو عملية أساسية في البيولوجيا الجزيئية. الشريط المقابل لـ DNA يُقرأ في الاتجاه المعاكس مع استبدال كل قاعدة بمتممها:

    • A ↔ T
    • G ↔ C
    def reverse_complement(dna_seq):
    """
    حساب المتمم العكسي لتسلسل DNA.

    المتمم العكسي مهم لأن DNA مزدوج الشريط،
    والشريط المقابل يُقرأ في الاتجاه المعاكس (3' إلى 5') لا تنسى تجريد الأحياء والرقمنة انعكاس للحقيقة.

    Args:
        dna_seq (str): تسلسل DNA يحتوي على A, T, G, C

    Returns:
        str: تسلسل المتمم العكسي

    Example:
        >>> reverse_complement("ATGC")
        'GCAT'
    """
    # الخطوة 1: تعريف جدول المتممات
    complement = {'A': 'T', 'T': 'A', 'G': 'C', 'C': 'G'}

    # الخطوة 2: استبدال كل قاعدة بمتممها
    comp_seq = ""
    for base in dna_seq:
        comp_seq = comp_seq + complement[base]

    # الخطوة 3: عكس الاتجاه
    rev_comp = comp_seq[::-1]

    return rev_comp

# اختبار
original = "ATGCGATCG"
result = reverse_complement(original)
print(f"الأصلي:        5'-{original}-3'")
print(f"المتمم العكسي: 3'-{result}-5'")
# الأصلي:        5'-ATGCGATCG-3'
# المتمم العكسي: 3'-CGATCGCAT-5'

    مبادئ كتابة دوال جيدة

    1. غرض واحد — كل دالة تؤدي مهمة واحدة محددة
    2. اسم وصفي — reverse_complement أوضح بكثير من rc
    3. توثيق — اكتب docstring يشرح ما تفعله الدالة ولماذا
    4. اختبار — تأكد أن الدالة تعطي نتائج صحيحة مع مدخلات معروفة

    دالة شاملة: إحصائيات التسلسل¶

    لنجمع ما تعلمناه في دالة واحدة مفيدة:

    def sequence_stats(dna_seq):
    """
    حساب إحصائيات شاملة لتسلسل DNA.

    Args:
        dna_seq (str): تسلسل DNA

    Returns:
        dict: قاموس يحتوي على الإحصائيات
    """
    length = len(dna_seq)

    # عد كل نيوكليوتيد
    counts = {
        'A': dna_seq.count('A'),
        'T': dna_seq.count('T'),
        'G': dna_seq.count('G'),
        'C': dna_seq.count('C')
    }

    # نسبة GC
    gc_content = (counts['G'] + counts['C']) / length * 100

    # نسبة AT
    at_content = (counts['A'] + counts['T']) / length * 100

    return {
        'length': length,
        'gc_content': round(gc_content, 2),
        'at_content': round(at_content, 2),
        'nucleotide_counts': counts
    }

# استخدام
stats = sequence_stats("ATGCGATCGTAGCTAGCT")
print(f"الطول: {stats['length']} bp")
print(f"محتوى GC: {stats['gc_content']}%")
print(f"محتوى AT: {stats['at_content']}%")
print(f"عدد النيوكليوتيدات: {stats['nucleotide_counts']}")

print(f"البروتين (حتى التوقف): {protein_to_stop}")  # MKA

    4. العمل مع تنسيقات الملفات البيولوجية¶

    لماذا تنسيقات مختلفة؟¶

    في الفصل الأول تعلمنا أن البيانات البيولوجية متنوعة. كل نوع بيانات يحتاج تنسيق يناسبه:

    التنسيق ماذا يحتوي متى يُستخدم
    FASTA التسلسل فقط + معرف مشاركة التسلسلات، قواعد البيانات
    FASTQ التسلسل + جودة كل قاعدة بيانات أجهزة التسلسل مباشرة
    GenBank التسلسل + تعليقات تفصيلية التسلسلات المرجعية المشروحة

    فكر فيها كمستويات من المعلومات: FASTA (الأبسط) → FASTQ (+ جودة) → GenBank (+ تعليقات كاملة)

    4.1 تنسيق FASTA¶

    الهيكل¶

    >seq_id وصف اختياري
ATGCGATCGTAGCTAGCTGATCGATCG
TCGATCGATCGTACGATCGATCGATCG
    • السطر الأول يبدأ بـ < ويحتوي معرف التسلسل والوصف
    • الأسطر التالية تحتوي التسلسل نفسه (قد يمتد على عدة أسطر)

    قراءة ملف FASTA¶

    from Bio import SeqIO

    • record.id — معرف التسلسل (الكلمة الأولى بعد >)
    • record.description — السطر الكامل بعد >
    • record.seq — كائن Seq يحتوي التسلسل
    • len(record) — طول التسلسل

def filter_by_length(input_file, output_file, min_length):

    الاحتفاظ بالتسلسلات الأطول من حد أدنى معين.

    from Bio import SeqIO

def analyze_fasta(input_file):
    """تحليل شامل لملف FASTA مع تقرير."""
    stats = []

    for record in SeqIO.parse(input_file, "fasta"):
        seq = str(record.seq)
        gc = (seq.count('G') + seq.count('C')) / len(seq) * 100
        stats.append({
            'id': record.id,
            'length': len(seq),
            'gc': gc
        })

    # طباعة تقرير
    print(f"\n{'='*60}")
    print(f"  تقرير تحليل: {input_file}")
    print(f"{'='*60}")
    print(f"  عدد التسلسلات: {len(stats)}")

    lengths = [s['length'] for s in stats]
    print(f"  أقصر تسلسل: {min(lengths)} bp")
    print(f"  أطول تسلسل: {max(lengths)} bp")
    print(f"  متوسط الطول: {sum(lengths)/len(lengths):.0f} bp")

    gc_values = [s['gc'] for s in stats]
    print(f"  متوسط GC: {sum(gc_values)/len(gc_values):.1f}%")
    print(f"{'='*60}\n")

    # طباعة تفاصيل كل تسلسل
    print(f"{'المعرف':<20} {'الطول':>8} {'GC%':>8}")
    print("-" * 40)
    for s in stats:
        print(f"{s['id']:<20} {s['length']:>8} {s['gc']:>7.1f}%")

# مثال
analyze_fasta("sequences.fasta")

    4.2 تنسيق FASTQ: التسلسل مع معلومات الجودة¶

    لماذا نحتاج الجودة؟¶

    أجهزة التسلسل ليست مثالية — كل قاعدة تُقرأ لها درجة ثقة وهي تعبير لجودة تحويل التسلسل إلى إشارة ضوئية ترمز للقاعدة النتروجينية. تنسيق FASTQ يضيف هذه المعلومة:

    @seq_id
ATGCGATCGTAGCT
+
IIIHHGGGFFFEEE
    • السطر 1: المعرف (يبدأ بـ @)
    • السطر 2: التسلسل
    • السطر 3: + (فاصل)
    • السطر 4: درجات الجودة — حرف لكل قاعدة

    فهم درجات الجودة (Phred Scores)¶

    كل حرف في سطر الجودة يمثل رقماً يعبر عن دقة القراءة:

    الحرف درجة Phred دقة القراءة معنى
    I 40 99.99% ممتاز
    F 37 99.98% جيد جداً
    5 20 99% مقبول
    ! 0 لا يوجد غير موثوق

    القاعدة العامة

    درجة Phred ≥ 30 تعني أن احتمال الخطأ أقل من 1 في 1000. هذا هو الحد الأدنى المقبول عادةً.

    قراءة وتحليل ملفات FASTQ¶

    قراءة أساسية:

    from Bio import SeqIO

for record in SeqIO.parse("reads.fastq", "fastq"):
    print(f"المعرف: {record.id}")
    print(f"التسلسل: {record.seq}")

    # درجات الجودة كأرقام
    qualities = record.letter_annotations["phred_quality"]
    print(f"الجودة: {qualities}")

    # حساب متوسط الجودة
    mean_quality = sum(qualities) / len(qualities)
    print(f"متوسط الجودة: {mean_quality:.1f}")
    print("---")

    تصفية القراءات حسب الجودة:

    from Bio import SeqIO

def filter_by_quality(input_fastq, output_fastq, min_quality=30):
    """
    الاحتفاظ بالقراءات التي متوسط جودتها أعلى من الحد الأدنى.

    هذه خطوة أساسية في أي خط تحليل للتسلسل (pipeline).
    القراءات منخفضة الجودة تُدخل أخطاء في التحليلات اللاحقة.
    """
    total = 0
    kept = 0

    high_quality_reads = []

    for record in SeqIO.parse(input_fastq, "fastq"):
        total += 1
        qualities = record.letter_annotations["phred_quality"]
        mean_qual = sum(qualities) / len(qualities)

        if mean_qual >= min_quality:
            high_quality_reads.append(record)
            kept += 1

    SeqIO.write(high_quality_reads, output_fastq, "fastq")

    removed = total - kept
    print(f"الإجمالي: {total} قراءة")
    print(f"تم الاحتفاظ: {kept} ({kept/total*100:.1f}%)")
    print(f"تم إزالة: {removed} ({removed/total*100:.1f}%)")

# مثال
filter_by_quality("raw_reads.fastq", "filtered_reads.fastq", min_quality=30)

    4.3 تنسيق GenBank: التسلسل المشروح بالكامل¶

    لماذا GenBank مختلف؟¶

    FASTA يعطيك التسلسل فقط. GenBank يعطيك كل شيء عن التسلسل:

    • التسلسل نفسه
    • الكائن الحي الذي ينتمي إليه
    • مواقع الجينات ومناطق الترميز
    • المراجع العلمية
    • التعليقات التوضيحية

    فكر في FASTA كنص خام، و GenBank ككتاب مع فهرس وهوامش وحواشي.

    قراءة ملف GenBank¶

    from Bio import SeqIO

# GenBank يحتوي عادةً على تسلسل واحد مشروح بالكامل

print(f"الكائن الحي: {record.annotations['organism']}")

    • SeqIO.read() — لملفات تحتوي تسلسلاً واحداً (تُرجع سجلاً واحداً)

    from Bio import Entrez, SeqIO

Entrez.email = "your.email@example.com"

def search_and_analyze(gene_name, organism="Homo sapiens"):
    """
    البحث عن جين في NCBI، تنزيل التسلسل، وتحليله.

    Args:
        gene_name: اسم الجين (مثل "BRCA1")
        organism: الكائن الحي (افتراضي: الإنسان)
    """
    print(f"\n🔍 البحث عن {gene_name} في {organism}...")

    # الخطوة 1: البحث
    handle = Entrez.esearch(
        db="nucleotide",
        term=f"{gene_name}[Gene] AND {organism}[Organism] AND mRNA[Filter]",
        retmax=1
    )
    results = Entrez.read(handle)
    handle.close()

    if not results['IdList']:
        print("  لم يُعثر على نتائج!")
        return

    seq_id = results['IdList'][0]
    print(f"  تم العثور على المعرف: {seq_id}")

    # الخطوة 2: الجلب
    handle = Entrez.efetch(
        db="nucleotide",
        id=seq_id,
        rettype="gb",
        retmode="text"
    )
    record = SeqIO.read(handle, "genbank")
    handle.close()

    # الخطوة 3: التحليل
    seq = str(record.seq)
    gc = (seq.count('G') + seq.count('C')) / len(seq) * 100

    print(f"  الوصف: {record.description}")
    print(f"  الطول: {len(seq):,} bp")
    print(f"  محتوى GC: {gc:.1f}%")

    # عد ميزات CDS
    cds_count = sum(1 for f in record.features if f.type == "CDS")
    print(f"  تسلسلات الترميز: {cds_count}")

# تحليل عدة جينات
for gene in ["BRCA1", "TP53", "EGFR"]:
    search_and_analyze(gene)

    6. أفضل الممارسات¶

    لماذا تهمنا "أفضل الممارسات"؟¶

    يمكنك كتابة برنامج يعمل لكنه فوضوي — ستعمل النتائج الآن، لكن بعد شهر لن تفهم ما كتبته. أو يأتي زميل ولا يستطيع استخدام كودك. أفضل الممارسات تجعل كودك:

    • مقروء — ستشكر نفسك في المستقبل :)
    • قابل لإعادة الاستخدام — اكتب مرة، استخدم مراراً
    • موثوق — يعمل بشكل صحيح ويمكن التحقق منه

    6.1 تنظيم النص البرمجي¶

    """
analyze_sequences.py - تحليل محتوى GC لتسلسلات FASTA

الاستخدام:
    python analyze_sequences.py input.fasta output.csv
"""

from Bio import SeqIO
import sys


def calculate_gc(sequence):
    """حساب نسبة محتوى GC لتسلسل."""
    gc_count = sequence.count('G') + sequence.count('C')
    return (gc_count / len(sequence)) * 100


def analyze_fasta(input_file, output_file):
    """
    تحليل محتوى GC لجميع التسلسلات وحفظ النتائج.

    Args:
        input_file: مسار ملف FASTA
        output_file: مسار ملف CSV للنتائج
    """
    results = []

    for record in SeqIO.parse(input_file, "fasta"):
        gc = calculate_gc(str(record.seq))
        results.append(f"{record.id},{len(record.seq)},{gc:.2f}")

    with open(output_file, 'w') as f:
        f.write("seq_id,length,gc_content\n")
        for line in results:
            f.write(line + "\n")

    print(f"تم تحليل {len(results)} تسلسل → {output_file}")


def main():
    """نقطة الدخول الرئيسية."""
    if len(sys.argv) != 3:
        print("الاستخدام: python analyze_sequences.py input.fasta output.csv")
        sys.exit(1)

    analyze_fasta(sys.argv[1], sys.argv[2])


if __name__ == "__main__":
    main()

    • الاستيرادات: كل المكتبات في البداية

    مستويات التوثيق

    1. Docstring للملف — ماذا يفعل البرنامج ككل

    6.3 اختبار الكود¶

    لا تثق بكودك حتى تختبره. اكتب اختبارات بسيطة للتأكد:

    def test_gc_content():
    """اختبار حساب محتوى GC بحالات معروفة."""

    # حالة 1: كل الحروف GC — يجب أن تكون 100%
    assert calculate_gc("GCGCGC") == 100.0

    # حالة 2: لا يوجد GC — يجب أن تكون 0%
    assert calculate_gc("ATATAT") == 0.0

    # حالة 3: نصف ونصف — يجب أن تكون 50%
    result = calculate_gc("ATGC")
    assert result == 50.0

    # حالة 4: تسلسل واقعي
    result = calculate_gc("ATGCGATCG")
    assert 55 < result < 56  # ~55.6%

    print("جميع الاختبارات نجحت!")

test_gc_content()

    assert تتحقق من شرط — إذا كان خاطئاً، يتوقف البرنامج ويخبرك بالخطأ.

    6.4 الأداء: التعامل مع الملفات الكبيرة¶

    كود فعال مقابل غير فعال

    المشكلة: ملفات التسلسل قد تحتوي ملايين السجلات. تحميلها كلها في الذاكرة مرة واحدة قد يستهلك ذاكرة الجهاز بالكامل.

    غير فعال — يحمل كل شيء في الذاكرة: 

    # ❌ خطير مع الملفات الكبيرة
all_records = list(SeqIO.parse("huge_file.fasta", "fasta"))
for record in all_records:
    process(record)

    فعال — يعالج تسلسلاً تلو الآخر: 

    # ✓ آمن — يحمل سجلاً واحداً في كل مرة
for record in SeqIO.parse("huge_file.fasta", "fasta"):
    process(record)

    الفرق: SeqIO.parse() تُرجع مكرراً (iterator) — تقرأ سجلاً واحداً في كل دورة وتتخلص منه قبل قراءة التالي. لكن عندما تحولها إلى قائمة بـ list()، تضع كل شيء في الذاكرة.
